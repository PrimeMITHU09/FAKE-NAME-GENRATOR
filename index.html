let generatedUsers = new Set(); // Toggle Dark / Day Mode function toggleTheme() { document.body.classList.toggle('dark-mode'); const btn = document.getElementById('theme-btn'); if (document.body.classList.contains('dark-mode')) { btn.innerHTML = ` Day Mode`; } else { btn.innerHTML = ` Night Mode`; } } function getRandomInt(min, max) { return Math.floor(Math.random() * (max - min + 1)) + min; } // দেশ অনুযায়ী নির্দিষ্ট জিপ কোড এবং কান্ট্রি কোড ম্যাপ const countryToNatMap = { "United States": { nat: "us", phoneCode: "1" }, "United Kingdom": { nat: "gb", phoneCode: "44" }, "Germany": { nat: "de", phoneCode: "49" }, "Australia": { nat: "au", phoneCode: "61" }, "France": { nat: "fr", phoneCode: "33" } }; async function generateIdentity() { let genderInput = document.getElementById("gender").value.toLowerCase(); let countryInput = document.getElementById("country").value; // জেন্ডার ফিল্টার let apiGender = genderInput === "random" ? (Math.random() > 0.5 ? "male" : "female") : genderInput; // কান্ট্রি ফিল্টার (সিলেক্ট করা কান্ট্রির সাথে API NAT ট্যাগ কানেক্ট করা) let selectedCountryData = countryToNatMap[countryInput] || { nat: "us", phoneCode: "1" }; let natCode = selectedCountryData.nat; try { // নির্বাচিত দেশের ডাটা আনতে nat=${natCode} পাঠানো হচ্ছে let response = await fetch(`https://randomuser.me/api/?gender=${apiGender}&nat=${natCode}`); let data = await response.json(); let user = data.results[0]; if (generatedUsers.has(user.login.uuid)) { return generateIdentity(); } generatedUsers.add(user.login.uuid); let firstName = user.name.first; let lastName = user.name.last; let middleInitial = String.fromCharCode(65 + Math.floor(Math.random() * 26)) + "."; let street = `${user.location.street.number} ${user.location.street.name}`; let city = user.location.city; let state = user.location.state; // জিপ কোড হ্যান্ডলিং (USA হলে বাধ্যতামূলকভাবে ৫ সংখ্যার ডিজিট জিপ কোড) let zip = user.location.postcode.toString(); if (countryInput === "United States") { // USA Zip Code ফরম্যাটিং (5 Digits) zip = zip.replace(/\D/g, ''); // অক্ষর থাকলে মুছে ফেলা if (zip.length < 5) { zip = getRandomInt(10000, 99999).toString(); } else if (zip.length > 5) { zip = zip.substring(0, 5); } } let photoUrl = user.picture.large; let phone = user.phone; let mmn = user.name.last + "son"; let ssn = `${getRandomInt(100, 999)}-${getRandomInt(10, 99)}-XXXX`; let geo = `${user.location.coordinates.latitude}, ${user.location.coordinates.longitude}`; let dobDate = new Date(user.dob.date); let monthNames = ["January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December"]; let birthday = `${monthNames[dobDate.getMonth()]} ${dobDate.getDate()}, ${dobDate.getFullYear()}`; let age = `${user.dob.age} years old`; // HTML Elements আপডেট করা document.getElementById("avatar-img").src = photoUrl; document.getElementById("fn").innerText = `${firstName} ${middleInitial}`; document.getElementById("ln").innerText = lastName; document.getElementById("street").innerText = street; document.getElementById("city").innerText = city; document.getElementById("state").innerText = state; document.getElementById("zip").innerText = zip; document.getElementById("name-meaning").innerText = firstName; document.getElementById("meaning-link").href = `https://www.behindthename.com/name/${firstName.toLowerCase()}`; document.getElementById("mmn").innerText = mmn; document.getElementById("ssn").innerText = ssn; document.getElementById("geo").innerText = geo; document.getElementById("phone").innerText = phone; document.getElementById("bday").innerText = birthday; document.getElementById("age").innerText = age; } catch (error) { console.error("Error generating identity: ", error); } } // পেজ লোড হওয়ার সাথে সাথে প্রথম ডাটা লোড হবে window.onload = function() { generateIdentity(); }; function copyText(elementId, btn) { const text = document.getElementById(elementId).innerText.trim(); navigator.clipboard.writeText(text).then(() => { const originalText = btn.innerText; btn.innerText = 'Copied!'; btn.classList.add('copied'); setTimeout(() => { btn.innerText = originalText; btn.classList.remove('copied'); }, 1500); }); }
